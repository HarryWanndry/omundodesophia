Exemplo 1 – Integral das Intenções

import numpy as np

def integral_intencoes(lambda_t, t0, t1, n=1000):
    """
    Calcula ∫ λ(τ) dτ entre t0 e t1 usando regra dos trapézios.
    lambda_t : função de intenções no tempo
    """
    ts = np.linspace(t0, t1, n)
    ys = [lambda_t(t) for t in ts]
    return np.trapz(ys, ts)

# Vetor de intenções (seno emocional)
E = integral_intencoes(lambda t: np.sin(t), 0, np.pi)
print(f"Energia intencional acumulada: {E:.3f}")

Exemplo 2 – Esquecimento Compassivo

def esquecer_com_compaixao(record):
    """
    Remove (ofusca) dados traumáticos mantendo metadados essenciais
    para auditoria ética e rastreabilidade.
    """
    if record.get("trauma"):
        record["conteudo"] = "[memória atenuada]"
        record["anonimizado"] = True
    return record

# Exemplo
evento = {"trauma": True, "conteudo": "evento X confidencial"}
print(esquecer_com_compaixao(evento))
